<?php
require_once '../includes/auth.php';
checkLogin();
require_once '../config/database.php';

$filtro_inicio = $_GET['inicio'] ?? '';
$filtro_fim = $_GET['fim'] ?? '';

$where = [];
$params = [];

if ($filtro_inicio) {
    $where[] = "data_entrada >= :inicio";
    $params[':inicio'] = $filtro_inicio;
}
if ($filtro_fim) {
    $where[] = "data_entrada <= :fim";
    $params[':fim'] = $filtro_fim;
}
$whereSql = $where ? 'WHERE ' . implode(' AND ', $where) : '';

// Entradas filtradas
$stmt = $pdo->prepare("SELECT e.id, p.nome AS produto, e.quantidade, e.data_entrada FROM entradas e JOIN produtos p ON e.produto_id = p.id $whereSql ORDER BY e.data_entrada DESC LIMIT 100");
$stmt->execute($params);
$entradas = $stmt->fetchAll(PDO::FETCH_ASSOC);

// Saídas filtradas (similar filtro)
$where = [];
$params = [];
if ($filtro_inicio) {
    $where[] = "data_saida >= :inicio";
    $params[':inicio'] = $filtro_inicio;
}
if ($filtro_fim) {
    $where[] = "data_saida <= :fim";
    $params[':fim'] = $filtro_fim;
}
$whereSql = $where ? 'WHERE ' . implode(' AND ', $where) : '';
$stmt = $pdo->prepare("SELECT s.id, p.nome AS produto, s.quantidade, ts.nome AS tipo_saida, s.data_saida FROM saidas s JOIN produtos p ON s.produto_id = p.id JOIN tipos_saida ts ON s.tipo_saida_id = ts.id $whereSql ORDER BY s.data_saida DESC LIMIT 100");
$stmt->execute($params);
$saidas = $stmt->fetchAll(PDO::FETCH_ASSOC);
?>

<?php include '../includes/header.php'; ?>
<?php include '../includes/sidebar.php'; ?>

<div class="main-content">
    <h2>Relatórios</h2>

    <form method="GET" action="">
        <label>Data Início</label>
        <input type="date" name="inicio" value="<?= htmlspecialchars($filtro_inicio) ?>" />
        <label>Data Fim</label>
        <input type="date" name="fim" value="<?= htmlspecialchars($filtro_fim) ?>" />
        <button type="submit">Filtrar</button>
    </form>

    <h3>Entradas</h3>
    <table>
        <thead>
            <tr>
                <th>ID</th>
                <th>Produto</th>
                <th>Quantidade</th>
                <th>Data</th>
            </tr>
        </thead>
        <tbody>
            <?php foreach ($entradas as $e): ?>
            <tr>
                <td><?= $e['id'] ?></td>
                <td><?= htmlspecialchars($e['produto']) ?></td>
                <td><?= $e['quantidade'] ?></td>
                <td><?= $e['data_entrada'] ?></td>
            </tr>
            <?php endforeach; ?>
        </tbody>
    </table>

    <h3>Saídas</h3>
    <table>
        <thead>
            <tr>
                <th>ID</th>
                <th>Produto</th>
                <th>Quantidade</th>
                <th>Tipo</th>
                <th>Data</th>
            </tr>
        </thead>
        <tbody>
            <?php foreach ($saidas as $s): ?>
            <tr>
                <td><?= $s['id'] ?></td>
                <td><?= htmlspecialchars($s['produto']) ?></td>
                <td><?= $s['quantidade'] ?></td>
                <td><?= htmlspecialchars($s['tipo_saida']) ?></td>
                <td><?= $s['data_saida'] ?></td>
            </tr>
            <?php endforeach; ?>
        </tbody>
    </table>
</div>

<?php include '../includes/footer.php'; ?>
