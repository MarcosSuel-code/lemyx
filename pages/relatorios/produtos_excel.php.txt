<?php
require_once '../../config/database.php';

header("Content-Type: application/vnd.ms-excel");
header("Content-Disposition: attachment; filename=relatorio_produtos.xls");

$produtos = $pdo->query("SELECT p.*, c.nome AS categoria_nome 
    FROM produtos p
    LEFT JOIN categorias c ON p.categoria = c.id")->fetchAll();

echo "<table border='1'>";
echo "<tr><th>Nome</th><th>Descrição</th><th>Categoria</th><th>Preço</th><th>Estoque</th></tr>";

foreach ($produtos as $p) {
    echo "<tr>";
    echo "<td>{$p['nome']}</td>";
    echo "<td>{$p['descricao']}</td>";
    echo "<td>{$p['categoria_nome']}</td>";
    echo "<td>R$ ".number_format($p['preco'],2,',','.')."</td>";
    echo "<td>{$p['estoque']}</td>";
    echo "</tr>";
}

echo "</table>";
?>
