<?php
session_start();

    header("Location: index.php");
    exit();

require_once 'config/database.php';

// Pegar lista de produtos para o select
$produtos = $pdo->query("SELECT id, nome, estoque FROM produtos ORDER BY nome")->fetchAll(PDO::FETCH_ASSOC);

// Tipos de saída (venda, perda, devolução, etc.)
$tipos_saida = ['Venda', 'Perda', 'Devolução', 'Outro'];

// Inserção da saída (se for POST)
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $produto_id = $_POST['produto_id'] ?? null;
    $quantidade = intval($_POST['quantidade'] ?? 0);
    $tipo_saida = $_POST['tipo_saida'] ?? null;
    $data_saida = $_POST['data_saida'] ?? date('Y-m-d');

    if ($produto_id && $quantidade > 0 && $tipo_saida) {
        // Verificar estoque suficiente
        $stmt = $pdo->prepare("SELECT estoque FROM produtos WHERE id = ?");
        $stmt->execute([$produto_id]);
        $produto = $stmt->fetch(PDO::FETCH_ASSOC);

        if ($produto && $produto['estoque'] >= $quantidade) {
            $pdo->beginTransaction();
            try {
                // Inserir saída
                $stmt = $pdo->prepare("INSERT INTO saida_mercadorias (produto_id, quantidade, tipo_saida, data_saida) VALUES (?, ?, ?, ?)");
                $stmt->execute([$produto_id, $quantidade, $tipo_saida, $data_saida]);

                // Atualizar estoque
                $stmt = $pdo->prepare("UPDATE produtos SET estoque = estoque - ? WHERE id = ?");
                $stmt->execute([$quantidade, $produto_id]);

                $pdo->commit();
                $mensagem = "Saída registrada com sucesso!";
            } catch (Exception $e) {
                $pdo->rollBack();
                $erro = "Erro ao registrar saída: " . $e->getMessage();
            }
        } else {
            $erro = "Estoque insuficiente para esse produto.";
        }
    } else {
        $erro = "Preencha todos os campos corretamente.";
    }
}
