<?php
session_start();
if (!isset($_SESSION['usuario'])) {
    header("Location: ../../index.php");
    exit();
}
require_once '../../config/database.php';

// Inserir saída
if (isset($_POST['produto_id'])) {
    // Verificar estoque disponível
    $stmt = $pdo->prepare("SELECT estoque FROM produtos WHERE id = ?");
    $stmt->execute([$_POST['produto_id']]);
    $produto = $stmt->fetch();

    if ($produto && $produto['estoque'] >= $_POST['quantidade']) {
        // Registrar saída
        $sql = "INSERT INTO saidas (produto_id, quantidade, tipo_saida, data) VALUES (?, ?, ?, NOW())";
        $stmt = $pdo->prepare($sql);
        $stmt->execute([
            $_POST['produto_id'],
            $_POST['quantidade'],
            $_POST['tipo_saida']
        ]);

        // Atualizar estoque
        $pdo->prepare("UPDATE produtos SET estoque = estoque - ? WHERE id = ?")
            ->execute([$_POST['quantidade'], $_POST['produto_id']]);

        header("Location: index.php");
        exit();
    } else {
        $erro = "Estoque insuficiente!";
    }
}

// Listar produtos e saídas
$produtos = $pdo->query("SELECT * FROM produtos")->fetchAll();
$saidas = $pdo->query("
    SELECT s.*, p.nome 
    FROM saidas s 
    JOIN produtos p ON s.produto_id = p.id
    ORDER BY s.data DESC
")->fetchAll();
